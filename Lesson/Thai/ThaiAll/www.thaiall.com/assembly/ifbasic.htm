    พื้นฐานการเลือกกระทำ
    

  1. รับค่าด้วย ah=01h รับและแสดงตัวอักษร
      .model  small
      .data
      .code
main  proc
      mov     ax,@data
      mov     ds,ax
wow:  mov     ah,01h ; get and display
      int     21h
      cmp     al,41h
      jne     wow
      mov     ah,4ch
      int     21h
main  endp
      .stack  200h    
      end     main

  2. รับค่าด้วย ah=08h แบบรับและไม่แสดงตัวอักษร
      .model  small
      .data
      .code
main  proc
      mov     ax,@data
      mov     ds,ax
wow:  mov     ah,08h ; get and not display
      int     21h
      mov     ah,02h ; display
      mov     dl,al
      int     21h
      cmp     al,41h
      jne     wow
bye:  mov     ah,4ch
      int     21h
main  endp
      .stack  200h    
      end     main
